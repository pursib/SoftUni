package Fundamentals.Regex;

import java.util.*;
import java.util.regex.Matcher;
import java.util.regex.Pattern;
import java.util.stream.Collectors;

public class Race {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);

        List<String> racers = Arrays.stream(scanner.nextLine().split(", ")).collect(Collectors.toList());
        String input = scanner.nextLine();
        String regexName = "[A-Z]*[a-z]*";
        String regexDistance = "[0-9]";
        Pattern patternName = Pattern.compile(regexName);
        Pattern patternDistance = Pattern.compile(regexDistance);

        Map<String, Integer> racersDistance = new LinkedHashMap<>();

        while (!input.equals("end of race")) {
            Matcher matcherName = patternName.matcher(input);
            Matcher matcherDistance = patternDistance.matcher(input);

            StringBuilder racerName = new StringBuilder();
            int distance = 0;
            while (matcherName.find()) {
                String name = matcherName.group();
                racerName.append(name);
            }
            while (matcherDistance.find()) {
                int currentNum = Integer.parseInt(matcherDistance.group());
                distance += currentNum;
            }

            String currentRacerName = racerName.toString();
            if (racers.contains(currentRacerName)) {

                if (!racersDistance.containsKey(currentRacerName)) {
                    racersDistance.put(currentRacerName, distance);
                } else {
                    racersDistance.put(currentRacerName, racersDistance.get(currentRacerName) + distance);
                }

            }


            input = scanner.nextLine();
        }
        List<String> firstThree = racersDistance.entrySet().stream().sorted(Map.Entry.comparingByValue(Comparator.reverseOrder()))
                .limit(3).map(Map.Entry::getKey).collect(Collectors.toList());
        System.out.println("1st place: " + firstThree.get(0));
        System.out.println("2nd place: " + firstThree.get(1));
        System.out.println("3rd place: " + firstThree.get(2));

    }
}
