package Fundamentals.BasicSyntax.BasicSyntaxExercise;

import java.util.Scanner;

public class StrongNumber {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String number = scanner.nextLine();
        int result = 0;
        for (int i = 0; i < number.length(); i++) {
            char symbol = number.charAt(i);
            int digit = Integer.parseInt(String.valueOf(symbol));


            result += factorialUsingForLoop(digit);
        }

        int numberAsInt = Integer.parseInt(number);
        if(result == numberAsInt){
            System.out.println("yes");
        }else{
            System.out.println("no");
        }
    }


        public static int factorialUsingForLoop (int n){
            int fact = 1;
            for (int i = 2; i <= n; i++) {
                fact = fact * i;
            }
            return fact;
        }
    }
